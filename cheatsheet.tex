\section{Cholesky-Zerlegung}
$\begin{bmatrix}
 g_{11}     & &  \\
 \vdots & \ddots & \\
 g_{gk1}      & \cdots & g_{kk}
\end{bmatrix}$ \\
\begin{tabular}{@{}l}
$g_{11} = \sqrt{a_{11}}$ \\
$g_{kk} = \sqrt{a_{kk} - \sum_{j=1}^{k-1} g_{kj}^2}$ \\
$g_{ik} = \frac{a_{ik} - \sum_{j=1}^{k-1} g_{ij}g_{kj} }{g_{kk}}$ \\
\end{tabular} \\

\section{dividierte Differenzen}
\begin{tabular}{@{}ll@{}}
$f[x_i]$ & $= f_i$ \\
$f[x_0,...,x_n]$ & $= \frac{ f[x_0,...,x_{n-1}] } {x_0 - x_n} $
\end{tabular}
